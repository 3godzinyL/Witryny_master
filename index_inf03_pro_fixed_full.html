JavaScript — Sekcja 2: Matematyka i data

2.1 Dwie liczby + operator (+, -, *, :) → wynik

const A = document.querySelector('#m_a');
const B = document.querySelector('#m_b');
const OP = document.querySelector('#m_op');
document.querySelector('#m_go').addEventListener('click', () => {
  const a = parseFloat(A.value), b = parseFloat(B.value);
  if (Number.isNaN(a) || Number.isNaN(b)) { alert('Podaj liczby'); return; }
  let out = 0;
  switch(OP.value){
    case '+': out = a + b; break;
    case '-': out = a - b; break;
    case '*': out = a * b; break;
    case '/': out = b === 0 ? 'dzielenie przez zero' : (a / b); break;
  }
  alert('Wynik: ' + out);
});

// najmniejsza, największa, średnia
function arrMin(arr){ return Math.min(...arr); }
function arrMax(arr){ return Math.max(...arr); }
function arrAvg(arr){ return arr.reduce((s,x)=>s+x,0) / (arr.length||1); }

// NWD (Euclid) i NWW
function gcd(a,b){ while(b!==0){ [a,b] = [b, a % b]; } return Math.abs(a); }
function lcm(a,b){ return Math.abs(a*b) / gcd(a,b); }

// inne przydatne
function randInt(min,max){ return Math.floor(Math.random()*(max-min+1))+min; }

3.5 Funkcje — przykłady ważne na INF.03

// deklaracja + wartość domyślna + typowy zwrot
function sum(a=0,b=0){ return a + b; }

// silnia iteracyjnie
function factorial(n){
  n = Math.max(0, Math.floor(n||0));
  let r=1; for(let i=2;i<=n;i++) r*=i; return r;
}

// test pierwszości (prosty)
function isPrime(n){
  n = Math.floor(n);
  if (n<2) return false;
  for(let i=2;i*i<=n;i++){ if(n%i===0) return false; }
  return true;
}

// arrow function
const double = x => x*2;
